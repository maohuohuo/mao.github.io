防抖和节流

mao 2021-09-27 10:18:35
Categories： Tags： js

1.为什么要防抖和节流

当用户在某段时间内重复触发某一事件，增加了服务器的压力

2.什么是防抖和节流

a. 防抖(debounce)

函数在一段时间内只执行最后一次。
例如在触发相应事件后，等待两秒后再执行相应的函数。而在等待的的这两秒中再次触发相同的事件时，从最后这次事件触发时间再往后推迟两秒再执行函数。

b.节流(throttle)

在n秒内只执行一次函数
例如在初始时事件触发了3次，则该3次事件分别在n秒，2n秒，3n秒执行

3.实现防抖和节流

a.防抖(debounce)

html代码

<div class="content">
    <div class="box" id="box"></div>
</div>

js代码

var count =0
var box=document.querySelector('#box')
box.innerHTML=count
box.onmousemove =debounce(add,300,true)
function add(e){
 //this需指向box
  console.log(this)
  //event事件指向
  console.log(e)
  box.innerHTML=count++
}
function debounce(func,time,immediate){
	var timeId
	return function(){
		//改变this指向
		var context = this
		//改变event事件指向
  		var arg=arguments
  		
		clearTimeout(timeId)
		// 是否立即执行
		if(immediate){
			//标记,默认不立即执行
  			var flag=!timeId
  			timeId=setTimeout(()=>{
  				timeId=null
  			},time)
  			if(flag) func.apply(context,arg)

		}else{
			timeId=setTimeout(function(){
				func.apply(context,arg)
			},time)
		}
		
	}
}
b.节流(throttle)

有两种方法：①事件戳、②定时器

</div>

js代码

var count=0
var box=document.querySelector('#box')
box.innerHTML=count
//box.onmousemove=throttle(add,2000)
//box.onmousemove=throttle1(add,2000)
box.onmousemove=throttle2(add,2000)
function add(e){
 //this需指向box
  console.log(this)
  //event事件指向
  console.log(e)
  box.innerHTML=count++
}

//使用时间戳（第一次触发，最后不会调用函数）
function throttle(func,time){
  var context,args
    //之前的时间戳
  var old=0
	return function(){
    //改变this指向
    context = this
    //改变event事件指向
    args=arguments
    
	 //获取当前时间戳
   var now=new Date().valueOf()
   if ((now-old)>time) {
    //立即执行
    func.apply(context,args)
    old=now
   }
		
	}
}

//使用定时器（第一次不会触发，最后一次会触发）
function throttle1(func,time){
  var context,args,timeId
  return function(){
    //改变this指向
    context = this
    //改变event事件指向
    args=arguments
     if (!timeId) {
        timeId=setTimeout(()=>{
            timeId=null
            func.apply(context,args)
        },time) 
      }
  }
}

//双剑合璧（第一次触发，最后一次离开也执行）
function throttle2(func,time){
  var context,args,timeId
  var old=0
  return function(){
   var now=new Date().valueOf()
   //改变this指向
    context = this
    //改变event事件指向
    args=arguments
    if (now-old>time) {
        //立即执行
        if (timeId) {
          clearTimeout(timeId)
          timeId

        }
        func.apply(context,args)
        old=now
     }else if (!timeId) {
       timeId=setTimeout(()=>{
          old=new Date().valueOf()
          timeId=null
          func.apply(context,args)
      },time) 
    }
     
  }
}
